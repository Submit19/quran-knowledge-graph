### 2. Debounce Timing

| Event                     | Delay (ms) |
|---------------------------|------------|
| **Keyup → API call**      | **300 ms** |
| **Blur (hide list)**      | 0 (immediate) |
| **Focus (re‑show list)**  | 0 (immediate) |

- A 300 ms debounce prevents excessive calls while the user is typing.
- Implemented via a lightweight utility (`setTimeout`/`clearTimeout`) in `search.js`.

### 3. Backend Endpoint

**URL**: `POST /api/concept_suggest`  
**Content‑Type**: `application/json`  

**Request Body**

```json
{
  "query": "partial user input"
}
```

**Response (200 OK)**

```json
{
  "suggestions": [
    {
      "canonical": "forgive",
      "variants": ["forgive", "forgiver", "forgiveness"],
      "verse_count": 225,
      "concept_id": "c_00123"
    }
    // … up to 10 items
  ]
}
```

- Returns up to 10 best‑matching concepts ordered by relevance.
- If no matches, `suggestions` is an empty array.

**Error handling** – on non‑200 responses, show a generic “Unable to fetch suggestions” toast.

### 4. Front‑end UX Flow

1. **User types** → after 300 ms debounce, `POST /api/concept_suggest` is called.
2. **Loading state** – while waiting, show a thin spinner inside the input (`.loading` class).
3. **Results**  
   - **Non‑empty**: Populate `#concept-suggestions` with `<li>` items.
   - **Empty**: Show a single disabled item:  
     `<li class="no-results" role="option">No concepts found</li>`
4. **Empty‑state (initial load)** – the suggestions list remains hidden (`.hidden` class).
5. **Keyboard navigation** – Arrow‑Up/Down moves focus; `Enter` selects the highlighted suggestion.
6. **Mouse interaction** – Clicking a suggestion selects it.
7. **Selection action** – On select, redirect to `/concept/<concept_id>` or fire a custom event (`conceptSelected`) that the main app listens to and loads the associated verses.

### 7. JavaScript Overview (`search.js`)

```js
const input = document.getElementById('concept-search-input');
const list  = document.getElementById('concept-suggestions');
let debounceTimer = null;
let highlightedIdx = -1;

input.addEventListener('keyup', e => {
  clearTimeout(debounceTimer);
  const query = e.target.value.trim();
  if (!query) { hideList(); return; }

  debounceTimer = setTimeout(() => fetchSuggestions(query), 300);
});

input.addEventListener('blur', () => setTimeout(hideList, 200));
input.addEventListener('focus', () => { if (list.children.length) showList(); });

function fetchSuggestions(q) {
  input.classList.add('loading');
  fetch('/api/concept_suggest', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ query: q })
  })
  .then(r => r.json())
  .then(data => renderSuggestions(data.suggestions))
  .catch(() => renderNoResults())
  .finally(() => input.classList.remove('loading'));
}

function renderSuggestions(items) {
  list.innerHTML = '';
  if (!items.length) {
    list.innerHTML = '<li class="no-results" role="option">No concepts found</li>';
  } else {
    items.forEach((c, i) => {
      const li = document.createElement('li');
      li.setAttribute('role', 'option');
      li.dataset.id = c.concept_id;
      li.dataset.index = i;
      li.innerHTML = `
        <strong>${c.canonical}</strong> (${c.variants.join('/')}, ${c.verse_count} verses)
      `;
      li.addEventListener('mousedown', () => selectConcept(c));
      list.appendChild(li);
    });
  }
  highlightedIdx = -1;
  showList();
}

function renderNoResults() {
  list.innerHTML = '<li class="no-results" role="option">Error fetching suggestions</li>';
  showList();
}

function showList() { list.classList.remove('hidden'); }
function hideList() { list.classList.add('hidden'); }

function selectConcept(concept) {
  const event = new CustomEvent('conceptSelected', { detail: concept });
  document.dispatchEvent(event);
  hideList();
  input.value = concept.canonical;
}
```
